#+begin_src bash

fur="$HOME/ch_mapro/mapro/analysis"
seq="$HOME/try_seqwin/mapro/analysis"

extract() {
    local file="$1"

    if [[ ! -f "$file" ]]; then
        echo "-|-"
        return
    fi

    sens=$(grep -i "sensitivity" "$file" | head -1 | grep -oE '[0-9]+(\.[0-9]+)?%?')
    spec=$(grep -i "specificity" "$file" | head -1 | grep -oE '[0-9]+(\.[0-9]+)?%?')

    [[ -z "$sens" ]] && sens="-"
    [[ -z "$spec" ]] && spec="-"

    echo "$sens|$spec"
}

echo '| Taxon | FUR SCOP Sens | FUR SCOP Spec | SeqWin SCOP Sens | SeqWin SCOP Spec |'
echo '|--------+---------------+---------------+------------------+------------------|'

for d in $(comm -12 \
    <(find "$fur" -mindepth 1 -maxdepth 1 -type d -printf "%f\n" | sort) \
    <(find "$seq" -mindepth 1 -maxdepth 1 -type d -printf "%f\n" | sort))
do
    IFS='|' read -r fur_sens fur_spec <<< "$(extract "$fur/$d/scop.out")"
    IFS='|' read -r seq_sens seq_spec <<< "$(extract "$seq/$d/scop.out")"

    if [[ "$fur_sens" == "-" &&
          "$fur_spec" == "-" &&
          "$seq_sens" == "-" &&
          "$seq_spec" == "-" ]]; then
        continue
    fi

    printf '| %s | %s | %s | %s | %s |\n' \
        "$d" \
        "$fur_sens" "$fur_spec" \
        "$seq_sens" "$seq_spec"
done
#+end_src
