number of markers and total lenght of the markers for each taxa
#+begin_src bash
  fur=~/ch_mapro/mapro/analysis
  seqwin=~/try_seqwin/mapro/analysis

  printf "%-8s %8s %10s %8s %10s\n" "Taxon" "Fur_n" "Fur_bp" "Seq_n" "Seq_bp"

  for taxon in $(ls "$fur"); do
      [ -f "$fur/$taxon/markers.fasta" ] && [ -f "$seqwin/$taxon/markers.fasta" ] || continue

      read fur_n fur_bp < <(awk '/^>/{n++;next}{bp+=length}END{print n,bp}' "$fur/$taxon/markers.fasta")
      read seq_n seq_bp < <(awk '/^>/{n++;next}{bp+=length}END{print n,bp}' "$seqwin/$taxon/markers.fasta")

      printf "%-8s %8d %10d %8d %10d\n" \
             "$taxon" "$fur_n" "$fur_bp" "$seq_n" "$seq_bp"
  done

#+end_src

#+begin_src bash
find . -name markers.fasta | sort | while read -r f; do
    echo "===== $f ====="
    awk '
    /^>/ {
        if (seen) print len
        seen = 1
        n++
        len = 0
        next
    }
    {
        len += length($0)
    }
    END {
        if (seen) print len
        printf("Total markers: %d\n\n", n)
    }
    ' "$f"
done
#+end_src
