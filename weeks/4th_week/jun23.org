  echo -e "strain\tparent_id\tparent_rank"

  while read line
  do
      code=$(echo "$line" | cut -f1)
      strain=$(echo "$line" | cut -f3)

      pid=$(taxi "$strain" neidb |
                tail -n 1 |
                awk '{print $2}')

      prank=$(ants $pid neidb |
                  tail -n 1 |
                  awk '{print $NF}')

      echo -e "$code\t$pid\t$prank"

  done < newlist.txt

#+end_src
