      for a in targets/*; do
  	r=$(grep $typeAcc $a)
  	if [[ $r != "" ]]; then
  	    leafLabel=$(basename $a)
  	    break
  	fi
      done
      
      if [[ $leafLabel != "" ]]; then
  	targetClade=$(fintac all.nwk |
  			  tail -n +2 |
  			  head -n 1 |
  			  awk '{print $1}')
  	t=$(climt $leafLabel all.nwk |
  		awk -v v=$targetClade '$2==v{print 2*($3+$4)}')
  	threshold=$t
  	r=$(grep '^False' scop.out)
  	if [[ $r != "" ]]; then
  	    cops -D -d $bdb -r $typeAcc -t $threshold \
  		 scop.out > cops.out
  	else
  	    cp scop.out cops.out
  	fi
      fi
  fi
