* Analyzing the target taxa

The accessions are contained in =acc.txt=. This file contains four
columns:

- GenBank accession
- Is target
- Is sampled (100 genomes)
- Is sampled (1000 genomes)

choose whether to analyze the 100-genome or the 1000-genome sample
by selecting the corresponding column.

first extract the accessions of the sampled genomes.

#+begin_src bash
SAMPLE_COL=4

awk -F '\t' -v col="$SAMPLE_COL" '
NR>1 && $col=="TRUE" {print $1}
' acc.txt > accessions.txt
#+end_src

download the sampled genomes in dehydrated form.

#+begin_src bash
datasets download genome accession \
    --inputfile accessions.txt \
    --dehydrated
#+end_src

We unzip and rehydrate the data.

#+begin_src bash
unzip -o ncbi_dataset.zip
datasets rehydrate --directory .
#+end_src

collect all genomes into a single directory.

#+begin_src bash
mkdir -p all

find ncbi_dataset/data -name "*_genomic.fna" | while read -r f
do
    acc=$(basename "$(dirname "$f")")
    ln -sf "$(realpath "$f")" "all/${acc}.fasta"
done
#+end_src

calculate the phylogeny of the sampled genomes.

#+begin_src bash
phylonium all/* \
    | nj \
    | midRoot \
    | land > tree.nwk
#+end_src

plot the phylogeny.

#+begin_src bash
plotTree tree.nwk
#+end_src

construct the directories containing the target and neighbor
genomes according to the annotations in =acc.txt=.

#+begin_src bash
mkdir -p targets neighbors

awk -F '\t' -v col="$SAMPLE_COL" '
NR>1 && $col=="TRUE" {
    print $1 "\t" $2
}
' acc.txt | while IFS=$'\t' read -r acc target
do
    if [ "$target" = "TRUE" ]; then
        ln -sf "$(pwd)/all/${acc}.fasta" "targets/${acc}.fasta"
    else
        ln -sf "$(pwd)/all/${acc}.fasta" "neighbors/${acc}.fasta"
    fi
done
#+end_src

construct the Fur database.

#+begin_src bash
/usr/bin/time -v -o makeFurDb.res \
makeFurDb \
    -T 1 \
    -t targets \
    -n neighbors \
    -o \
    -d target.db
#+end_src

run Fur on the database.

#+begin_src bash
/usr/bin/time -v -o fur.res \
fur -m -d target.db > target.fasta
#+end_src
